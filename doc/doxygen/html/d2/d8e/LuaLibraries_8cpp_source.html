5 unsigned int BBP::Services::ApplicationServices::Lua::fetchIndexForLuaInstance(lua_State *L)
6 {
7  // ! Checking the argument for nullptr has been omitted on purpose !
8  // This is because this function can then also find any applications which have a nullptr, IE invalid applications. They could then be cleaned up.
9 
10  // Loop over each registered application and return the one whose applicationState matches the parameter L
11  for (int i = 0; i < BBP::Services::ApplicationServices::applicationsCount; i++)
12  {
13  // If the application does not exist, continue.
14  if (BBP::Services::ApplicationServices::applicationsVector[i] == nullptr)
15  continue;
16 
17  // If the applicationState matches the parameter, return this index.
18  if (BBP::Services::ApplicationServices::applicationsVector[i]->applicationState == L)
19  return i;
20  }
21 
22  // Not a single application exists with this lua environment. Return an invalid argument.
23  return 0xFFFF; // The caller should check if this specific value is returned.
