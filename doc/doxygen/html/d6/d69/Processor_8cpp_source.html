222 void BBP::esa::esaProcessor::finalize()
223 {
224 
225  // Set executable flags
226  application->builder.segments[application->builder.textSegment].flags(0x1);
227  application->builder.segments[application->builder.handlerSegment].flags(0x1);
228  application->builder.segments[application->builder.gotSegment].flags(0x1);
229 
230  // Get executable segment sizes
231  std::size_t textSize = application->builder._text().stack.atElement;
232  std::size_t handlerSize = application->builder._handlers().size();
233  std::size_t gotSize = application->builder._got().size();
234 
235  // Handler offsets is on physical address 0 because the thread will then always know how to handle errors.
236  // After that comes the got. This is so that the thread knows that the GOT is always at physical address 0+255.
237  // After that comes .text
238  application->builder.segments[application->builder.handlerSegment].paddr(0);
239  application->builder.segments[application->builder.gotSegment].paddr(handlerSize);
240  application->builder.segments[application->builder.textSegment].paddr(handlerSize + gotSize);
241 
242  // For virtual memory it is the exact opposite. First .text, then .got, then .handlers. This is because
243  // keeping .text at '0' means that the page stuff emitted by elsa is consistent with this virtual memory
244  std::address_t executablePage = application->builder.text << 16;
245  application->builder.segments[application->builder.textSegment].vaddr(executablePage);
246  application->builder.segments[application->builder.gotSegment].vaddr(executablePage + textSize);
247  application->builder.segments[application->builder.handlerSegment].vaddr(executablePage + textSize + gotSize);
248 
249  // Also update entry point if set
250  if (application->builder.entry() || application->builder.type() == std::ELF::ET_EXEC)
251  application->builder.entry(application->builder.entry() + executablePage);
252 
253  // Now set sizes
254  application->builder.segments[application->builder.textSegment].filesz(textSize);
255  application->builder.segments[application->builder.textSegment].memsz(textSize);
256 
257  application->builder.segments[application->builder.gotSegment].filesz(gotSize);
258  application->builder.segments[application->builder.gotSegment].memsz(gotSize);
259 
260  application->builder.segments[application->builder.handlerSegment].filesz(handlerSize);
261  application->builder.segments[application->builder.handlerSegment].memsz(handlerSize);
262 
263  // Now set offsets
264  application->builder.segments[application->builder.textSegment].offset(application->builder._text().offset());
265  application->builder.segments[application->builder.handlerSegment].offset(application->builder._handlers().offset());
266  application->builder.segments[application->builder.gotSegment].offset(application->builder._got().offset());
267 
268  // Get sizes of .data, .rodata, and .bss
269  std::size_t dataSize = application->builder._data().stack.atElement;
270  std::size_t rodataSize = application->builder._rodata().stack.atElement;
271  std::size_t bssSize = application->builder._bss().stack.atElement;
272 
273  std::address_t dataPage = application->builder.data << 16;
274  std::address_t rodataPage = application->builder.rodata << 16;
275  std::address_t bssPage = application->builder.bss << 16;
276 
277  // Next set .data, .rodata, .bss
278  application->builder.segments[application->builder.dataSegment].filesz(dataSize);
279  application->builder.segments[application->builder.dataSegment].memsz(dataSize);
280 
281  application->builder.segments[application->builder.rodataSegment].filesz(rodataSize);
282  application->builder.segments[application->builder.rodataSegment].memsz(rodataSize);
283 
284  application->builder.segments[application->builder.bssSegment].filesz(0);
285  application->builder.segments[application->builder.bssSegment].memsz(bssSize);
286

288  application->builder.segments[application->builder.dataSegment].flags(0b110);
289  application->builder.segments[application->builder.rodataSegment].flags(0b100);
290  application->builder.segments[application->builder.bssSegment].flags(0b110);
291 
292  // Now set virtual memories
293  application->builder.segments[application->builder.dataSegment].vaddr(dataPage);
294  application->builder.segments[application->builder.rodataSegment].vaddr(rodataPage);
295  application->builder.segments[application->builder.bssSegment].vaddr(bssPage);
296 
297  // Set offsets
298  application->builder.segments[application->builder.dataSegment].offset(application->builder._data().offset());
299  application->builder.segments[application->builder.rodataSegment].offset(application->builder._rodata().offset());
300  application->builder.segments[application->builder.bssSegment].offset(application->builder._bss().offset());
301 
302  // Set physical addresses. First comes rodata, then data, then bss
303  std::index_t executableSize = handlerSize + gotSize + textSize;
304  application->builder.segments[application->builder.dataSegment].paddr(executableSize + rodataSize);
305  application->builder.segments[application->builder.rodataSegment].paddr(executableSize);
306  application->builder.segments[application->builder.bssSegment].paddr(executableSize + rodataSize + dataSize);
307

309  application->builder.segments[application->builder.dataSegment].type(std::ELF::PT_TLS);
310  application->builder.segments[application->builder.bssSegment].type(std::ELF::PT_TLS);
