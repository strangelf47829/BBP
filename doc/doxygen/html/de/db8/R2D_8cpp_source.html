147 void R2D::GetFontFromROM(window *window, int offset_start, int nums)
148 {
149  // Get Font from Memory. This returns a calloc-ed pointer to the specified chunk of memory.
150  // Since this pointer is c-alloced, and the function doesn't touch it, it is safe to just store into the window font.
151  int *font = BBP::IO::Memory::ROM::readROMIntChunk(offset_start, offset_start + nums * 2 - 1);
152  mem_t *mapping = BBP::IO::Memory::ROM::readROMChunk((offset_start + nums * 2) * 4, ((offset_start + nums * 2) * 4) + nums * 8);
153 
154  // If window pointer is invalid, something went wrong. Throw BIOS error for now to test functionality
155  if (window == nullptr)
156  BBP::Services::Throw(KERNEL_BIOSERROR);
157 
158  // Since the window may have had a previous font placed in it (which was c-alloced), overriding it would cause memory leak.
159  // This checks if the font pointer is a nullptr, and if not (meaning it holds a font), release the c-alloced data.
160  if (window->typeFace != nullptr)
161  BBP::Services::Free(window->typeFace);
162 
163  if (window->charMapping != nullptr)
164  BBP::Services::Free(window->charMapping);
165 
166  // Now write the font mapping to the window font mapping
167  // TODO inefficient, optimize later though
168 
169  // First find the \0 character from ROM
170  int nullCharacterIndex = 0;
171  for (int i = 0; i < nums; i++)
172  {
173  if (mapping[i]) // If the character is not '\0', continue
174  continue;
175  nullCharacterIndex = i; // \0 found!
176  break;
177  }
178  // Then write that font map to each character
179  for (int i = 0; i < 256; i++)
180  window->fontMapping[i] = nullCharacterIndex;
181 
182  // Then overwrite the known fonts from the map
183  for (int i = 0; i < nums; i++)
184  window->fontMapping[mapping[i]] = i;
185 
186  // Now the window font can be overriden without causing memory leak.
187  window->typeFace = font;
188  window->charMapping = mapping;
189 
190 }

315 {

317

319  // draw backwards
320  if (steep)
321  {

323  swap(&x1, &y1);
324  }
325  if (x0 > x1)
326  {
327  swap(&x0, &x1);
328  swap(&y0, &y1);
329  }
330 
331  //compute the slope
332  float dx = x1 - x0;
333  float dy = y1 - y0;
334  float gradient = dy / dx;
335  if (dx == 0.0)
336  gradient = 1;
337

339  int xpxl2 = x1;
340  float intersectY = y0;
341 
342  // main loop
343  if (steep)
344  {
345  int x;
346  for (x = xpxl1; x <= xpxl2; x++)
347  {
348  // pixel coverage is determined by fractional

350  R2D::setPixel(window, iPartOfNumber(intersectY), x);
351  R2D::setPixel(window, iPartOfNumber(intersectY) - 1, x);
